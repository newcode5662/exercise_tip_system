        }

        }
      };
      
      request.onsuccess = (event) => {
        this.db = (event.target as IDBOpenDBRequest).result;
        resolve();
      };
      
      request.onerror = (event) => {
        reject((event.target as IDBOpenDBRequest).error);
      };
    });
  }

  async addWorkoutLog(log: WorkoutLog): Promise<void> {
    return new Promise((resolve, reject) => {
      if (!this.db) return reject(new Error('Database not initialized'));
      
      const transaction = this.db.transaction(['workout_logs'], 'readwrite');
      const store = transaction.objectStore('workout_logs');
      const request = store.add(log);
      
      request.onsuccess = () => resolve();
      request.onerror = () => reject(request.error);
    });
  }

  async getRecentWorkoutLogs(days: number = 30): Promise<WorkoutLog[]> {
    return new Promise((resolve, reject) => {
      if (!this.db) return reject(new Error('Database not initialized'));
      
      const transaction = this.db.transaction(['workout_logs'], 'readonly');
      const store = transaction.objectStore('workout_logs');
      const index = store.index('date');
      const request = index.getAll();
      
      request.onsuccess = () => {
        const logs = request.result as WorkoutLog[];
        const cutoffDate = new Date();
        cutoffDate.setDate(cutoffDate.getDate() - days);
        
        const filtered = logs.filter(log => 
          new Date(log.date) >= cutoffDate
        ).sort((a, b) => 
          new Date(b.date).getTime() - new Date(a.date).getTime()
        );
        
        resolve(filtered);
      };
      
      request.onerror = () => reject(request.error);
    });
  }

const workoutStore = useWorkoutStore()

const formattedDate = computed(() => {

  const lastLog = recentLogs.value[0]
  if (!lastLog) return false
  
  const lastDate = new Date(lastLog.date)
  const today = new Date()
  const diffDays = Math.floor((today.getTime() - lastDate.getTime()) / (1000 * 3600 * 24))
